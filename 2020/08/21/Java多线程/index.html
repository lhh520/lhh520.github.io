2.实现Runnable接口(implements)
实现Runnable接口 重写run()方法 主函数Thred.start() 避免单继承局限性 方便同一个对象被多个线程使用


public class test_Runnable implements Runnable
{
    public void run()//重写run()方法
    {
        for (int i = 0; i < 20; i++)
        {
            System.out.println("多线程");
        }
    }

    public static void main(String[] args)
    {
        test_Runnable t1=new test_Runnable();//实现类对象
        Thread tt1=new Thread(t1);//创建线程
        tt1.start();
        for (int i = 0; i < 200; i++) {
            System.out.println("主程序");
        }
    }
}

4.多线程——龟兔赛跑


public class test_run implements Runnable {
    private static String winner;

    public void run()
    {
        for (int i = 0; i < 101; i++)
        {
            boolean flag = gameover(i);
            if (flag)
                break;

            System.out.println(Thread.currentThread().getName() + "-->跑了" + i + "步");
        }
    }

    public boolean gameover(int steps)
    {
        if (winner != null)//已经存在胜利者了
            return true;
        {
            if (steps >= 100) {
                winner = Thread.currentThread().getName();
                System.out.println("winner is " + winner);
                return true;
            }
        }
        return false;
    }


    public static void main(String[] args) {
        test_run t1=new test_run();
        new Thread(t1,"兔子").start();
        new Thread(t1,"乌龟").start();
    }

}

public class test_Thread_stop implements Runnable
{   private boolean flag=true;//设置标志位
    public void run()
    {
        int i=0;
        while(flag)//不断进行查询，看线程是否需要停止
            System.out.println("run... Thread"+ i++);
    }
    //设置公开方法停止线程，转换标志位
    public void stop()

        this.flag=false;

            if(i==900)
                tsp.stop();

    }
}

5.6 守护线程
用户线程： main 必须执行完才可以停止程序
守护线程： gc(垃圾回收机制) 不用等到执行完就可以停止程序


public class test_Thread_Daemon
{
    public static void main(String[] args) {
        Dae dd=new Dae();
        costom ss=new costom();
        Thread tt=new Thread(dd);
        tt.setDaemon(true);//设置为守护进程
        tt.start();

        new Thread(ss).start();
    }
}
class Dae implements Runnable
{
    @Override
    public void run() {
        while(true)
        {
            System.out.println("守护线程");
        }
    }
}
class costom implements Runnable
{
    @Override
    public void run() {
        for (int i = 0; i < 200; i++) {
            System.out.println("用户进程 "+i);
        }
        System.out.println("=====goodbye======");
    }
}

7.线程池
背景：经常创建和销毁，使用特别大的资源，并发情况下的线程，对性能影响比较大，可以提前创建好多个线程，放入线程池中，使用时直接获取，使用完放回池子中，避免重复的创建和销毁。


import java.util.concurrent.Executor;
import java.util.concurrent.ExecutorService;
import java.util.concurrent.Executors;

public class Testpool
{
    public static void main(String[] args) {
        //创建服务，创建线程池
        //newFixedThreadPool后面跟着池子大小
        ExecutorService es2 = Executors.newFixedThreadPool(10);
        es2.execute(new MyThread());
        es2.execute(new MyThread());
        es2.execute(new MyThread());
        es2.execute(new MyThread());
        es2.execute(new MyThread());
        //线程池关闭
        es2.shutdown();
    }
}
class MyThread implements Runnable
{
    @Override
    public void run() {

            System.out.println(Thread.currentThread().getName());

    }
}
