2.主函数类继承Thread，重写run()方法，赋值网址和名字两个参数，线程类启动


import org.apache.commons.io.FileUtils;

import java.io.File;
import java.io.IOException;
import java.net.URL;

//练习Thread 实现多线程下载图片
public class test_Thread extends Thread
{
     private String url;
     private String name;
     public test_Thread(String url,String name)
     {
         this.url=url;
         this.name=name;
     }
     @Override //@Override:只能用在方法之上的，用来告诉别人这一个方法是改写父类的
     public void run()//重新run方法
     {
         WebDownLoader web=new WebDownLoader();
         web.downloader(url,name);
         System.out.println("下载了文件名："+name);
     }

    public static void main(String[] args)
    {   //https://img-blog.csdnimg.cn/20190119210318665.png
        test_Thread thread1=new test_Thread("https://img-blog.csdnimg.cn/20190119210318665.png","1.png");
        test_Thread thread2=new test_Thread("https://img-blog.csdnimg.cn/20190122113229685.jpg","2.jpg");
        test_Thread thread3=new test_Thread("http://u.thsi.cn/imgsrc/flashcms/502757715_da4b3b266c45448da28571cac05a5784.jpg","3.jpg");
        thread1.start();
        thread2.start();
        thread3.start();
    }

}
class WebDownLoader
{
    public void downloader(String url,String name)
    {   try{
        FileUtils.copyURLToFile(new URL(url),new File(name));}//将url地址下载下来成为文件
        catch(IOException e)
        {
            e.printStackTrace();
            System.out.println("IO异常，下载器有问题");
        }
    }
}

7.线程池
背景：经常创建和销毁，使用特别大的资源，并发情况下的线程，对性能影响比较大，可以提前创建好多个线程，放入线程池中，使用时直接获取，使用完放回池子中，避免重复的创建和销毁。


import java.util.concurrent.Executor;
import java.util.concurrent.ExecutorService;
import java.util.concurrent.Executors;

public class Testpool
&#123;

        //newFixedThreadPool后面跟着池子大小

        es2.execute(new MyThread());
        es2.execute(new MyThread());
        es2.execute(new MyThread());
        es2.execute(new MyThread());
        es2.execute(new MyThread());
